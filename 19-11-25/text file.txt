package FIRSTCLASS;
public class OMR {
	public static void main(String[] args) {

		Student s=new Student();
		s.name();
		Placement p =new Placement();
		p.fees();
		p.subject();
		Faculty f=new Faculty();
		f.department();
	}
}
class University{
	void department(){
		System.out.println("Department: DataScience");
	}
}

class Faculty extends University{
	void subject(){
		System.out.println("Subject: Java Full Stack");
	}
}

class Student extends Faculty{
	void name(){
		System.out.println("Name: XXXX");
	}
}

class Placement extends Student{
	void fees(){
		System.out.println("Fees: 30000");
	}
}
