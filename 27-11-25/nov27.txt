package package1;

import java.util.*;

public class lists {

	public static void main(String[] args) {

		ArrayList<Object> arrayList = new ArrayList<>();

		arrayList.add(1);
		arrayList.add("Java");
		arrayList.add(20.11);
		arrayList.add('J');
		arrayList.add(true);

		LinkedList<Object> linkedList = new LinkedList<>();

		linkedList.add(2);
		linkedList.add("FullStack");
		linkedList.add(20.22);
		linkedList.add('F');
		linkedList.add(false);

		System.out.println("ArrayList Elements:");
		Iterator<Object> itr1 = arrayList.iterator();
		while (itr1.hasNext()) {
			System.out.println(itr1.next());
		}

		System.out.println("\nLinkedList Elements:");
		Iterator<Object> itr2 = linkedList.iterator();
		while (itr2.hasNext()) {
			System.out.println(itr2.next());
		}
	}

}

2. LINKED LIST EXAMPLE TO ADD ELEMENTS

package package1;

import java.util.*;

public class lists {

    public static void main(String[] args) {

        LinkedList<String> link = new LinkedList<String>();
        link.add("AAA");
        link.add("BBB");
        link.add("CCC");
        System.out.println("Initial List:");
        System.out.println(link);
        System.out.println();

        link.add(1, "DDD");
        System.out.println("Adding an element at the specific position:");
        System.out.println(link);
        System.out.println();

        LinkedList<String> link2 = new LinkedList<String>();
        link2.add("EEE");
        link2.add("FFF");
        System.out.println("elements of list 2:");
        System.out.println(link2);
        System.out.println();

        link.addAll(link2);
        System.out.println("Adding second list elements to the first list: ");
        System.out.println(link);
        System.out.println();

        LinkedList<String> link3 = new LinkedList<String>();
        link3.add("GGG");
        link3.add("HHH");
        System.out.println("elements of list 3:");
        System.out.println(link3);
        System.out.println();

        link.addAll(1, link3);
        System.out.println("Adding list 3 elements to list 1 at specific position: ");
        System.out.println(link);
        System.out.println();

        link.addFirst("III");
        System.out.println("Adding an element at the first position:");
        System.out.println(link);
        System.out.println();

        link.addLast("JJJ");
        System.out.println("Adding an element at the last position");
        System.out.println(link);
        
        Iterator i=link.descendingIterator();  
        while(i.hasNext())  
        {  
            System.out.println(i.next());  
        }  


    }
}



3. LINKED LIST EXAMPLE TO DELETE ELEMENTS


package package1;

import java.util.*;

public class lists {

    public static void main(String[] args) {
         LinkedList<String> list1 = new LinkedList<String>();  
         
         list1.add("AAA");  
         list1.add("BBB");  
         list1.add("CCC");  
         list1.add("DDD");  
         list1.add("EEE");  
         list1.add("FFF");  
         list1.add("GGG");  
         list1.add("EEE");     // duplicate
         list1.add("FFF");     // duplicate
         list1.add("HHH");  
         
         System.out.println("Initial list of elements: ");
         System.out.println(list1);
         System.out.println();
         
         list1.remove("BBB");  
         System.out.println("Removing specific element from list:"); 
         System.out.println(list1);
         System.out.println();
    
         list1.remove(0);  
         System.out.println("Removing element on specific position:");  
         System.out.println(list1);
         System.out.println();
         
         LinkedList<String> list2 = new LinkedList<String>();
         list2.add("AAA");  
         list2.add("III");   
         
         list1.addAll(list2);  
         System.out.println("Adding list 2 to list 1:");   
         System.out.println(list1);
         System.out.println();
    
         list1.removeAll(list2);  
         System.out.println("Removing all elements in list 2:"); 
         System.out.println(list1);
         System.out.println();
     
         list1.removeFirst();  
         System.out.println("Removing first element from the list 1:"); 
         System.out.println(list1);
         System.out.println();
    
         list1.removeLast();  
         System.out.println("Removing last element from the list 1:");  
         System.out.println(list1);
         System.out.println();
         
         list1.removeFirstOccurrence("EEE");  
         System.out.println("Removing first occurrence of element from the list 1:");  
         System.out.println(list1);
         System.out.println();
         
         list1.removeLastOccurrence("FFF");  
         System.out.println("Removing last occurrence of element from the list 1:");  
         System.out.println(list1);
         System.out.println();
              
         list1.clear();  
         System.out.println("Removing all the elements available in the list:"); 
         System.out.println(list1);
    }
}


4. CREATE TREEST & HASHMAP

package package1;
import java.util.*;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.TreeSet;


public class lists {

    public static void main(String[] args) {
    		
    	ArrayList<String> list = new ArrayList<>();
        list.add("Mango");
        list.add("Apple");
        list.add("Banana");
        list.add("Apple");  
        list.add("Grapes");
        System.out.println("ArrayList (with duplicates):Â ");
        System.out.println(list);
        System.out.println();
        
//TREE SET
    		TreeSet<String> set = new TreeSet<>(list);
    		System.out.println("TreeSet (sorted, no duplicates): " + set);
        System.out.println();


    		set.add("DDD");
    		System.out.println("After adding 'DDD' to TreeSet: " + set);            
    		System.out.println();


    		System.out.println("Displaying TreeSet using loop:");
    		for (String item : set) {
    			System.out.println(item);

 // HASH MAP
    			HashMap<Integer, String> map = new HashMap<>();
    			map.put(101, "AAA");
    			map.put(102, "BBB");
    			map.put(103, "CCC");

    			System.out.println("HashMap: " + map);

    			System.out.println("Value for key 102: " + map.get(102));

    			map.put(102, "DDD");  
    			System.out.println("After update: " + map);

    			map.remove(103);
    			System.out.println("After removing key 103: " + map);

    			System.out.println("Keys: " + map.keySet());
    			System.out.println("Values: " + map.values());

    			System.out.println("Using loop:");
    			for (Integer key : map.keySet()) {
    				System.out.println(key + " = " + map.get(key));
    			}
    		}
    }
}
